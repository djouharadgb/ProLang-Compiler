BeginProject Test_branches;

Setup :
    define i : integer;
    define j : integer;
    define res : integer;

Run :
{
    res <- 0;

    %% Boucle externe : i de 0 a 2
    for i in 0 to 2
    {
        %% Boucle interne : j de 0 a 3
        for j in 0 to 3
        {
            res <- res + 1;
        }
        endfor;
    }
    endfor;

    out(res);
}

EndProject;
